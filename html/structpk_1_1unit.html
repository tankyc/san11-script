◆ is_alive()

virtual bool pk::unit::is_alive ( )
virtual
if (!is_instance(pk::unit::type_id))
return false;
if (!pk::is_alive(pk::get_person(leader)))
return false;
if (!pk::is_valid_person_id(deputy[0]) and deputy[0] > 0)
return false;
if (!pk::is_valid_person_id(deputy[1]) and deputy[1] > 0)
return false;
return true;
