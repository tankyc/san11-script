if (!is_instance(pk::person::type_id))
return false;
if (!pk::is_valid_mibun(mibun))
return false;
// 이 값은 剧本不可访问하다.
if (is_event_npc)
return true;
if (minun == 신분_미등장 or mibun == 신분_사망)
return false;
return true;
